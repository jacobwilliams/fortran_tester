135  class(tester_t), intent(in) :: this
136  logical, intent(in), optional :: errorstop
137 
138  logical :: do_errorstop

144 
145  write(*,*) 'fortran_tester:', this% n_errors, ' error(s) for', this% n_tests, 'test(s)'
